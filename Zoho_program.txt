#include<stdio.h>
void main()
{
int m=5,i,j;
while(m!=0)
{

   for(i=5;i>=1;i--)
      {
     if(i==m)
        {
         for(j=0;j<i;j++)
            {
             printf("%d",i);
             goto front;
            }
         }else
         {
         printf("%d",i);
         }
      }
       front:
       printf("\n");
       m--;
}

}
